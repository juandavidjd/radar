CREATE TABLE resultados_nueva (
    loteria TEXT NOT NULL,
    fecha TEXT NOT NULL,
    numero TEXT NOT NULL,
    signo TEXT DEFAULT '',
    PRIMARY KEY (loteria, fecha, numero, signo)
);

INSERT INTO resultados_nueva (loteria, fecha, numero, signo)
SELECT
    loteria,
    fecha,
    substr('0000' || CAST(numero AS INTEGER), -4, 4),
    IFNULL(signo, '')
FROM resultados
WHERE numero IS NOT NULL
  AND TRIM(numero) != ''
  AND numero GLOB '[0-9]*';

-- 3. Eliminar la tabla original
DROP TABLE resultados;

-- 4. Renombrar la tabla nueva
ALTER TABLE resultados_nueva RENAME TO resultados;
